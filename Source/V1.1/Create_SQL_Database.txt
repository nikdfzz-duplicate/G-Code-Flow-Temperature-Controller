CREATE TABLE IF NOT EXISTS EXTRUDER
(
  EXTRUDER_NAME VARCHAR(100) PRIMARY KEY NOT NULL,
  TEMP_RISE INTEGER NOT NULL,
  TEMP_RISE_TIME INTEGER NOT NULL,
  TEMP_FALL INTEGER NOT NULL,
  TEMP_FALL_TIME INTEGER NOT NULL,
  A_M_SMOOTH INTEGER NOT NULL,
  START_MACRO VARCHAR(50),
  MACRO_EXTRUDER VARCHAR(50),
  PRINTER_CONFIG VARCHAR(1000)
)

CREATE TABLE IF NOT EXISTS FILAMENT
(
  EXTRUDER_NAME VARCHAR(100) NOT NULL,
  FILAMENT_NAME VARCHAR(100) PRIMARY KEY NOT NULL,
  FILAMENT_TYPE VARCHAR(20) NOT NULL,
  HEIGHT_TEMP INTEGER NOT NULL,
  MID_TEMP INTEGER NOT NULL,
  LOW_TEMP INTEGER NOT NULL,
  HEIGHT_FLOW INTEGER NOT NULL,
  MID_FLOW INTEGER NOT NULL,
  LOW_FLOW INTEGER NOT NULL,
  HEIGHT_PA FLOAT,
  MID_PA FLOAT,
  LOW_PA FLOAT,
  ADJUST_PA BOOL,
  SPEED_QUALITY_OPT INTEGER
)
